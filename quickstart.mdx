## Get Started in Three Steps

<Steps>
  <Step title="Get Payment Proof">
    Connect your wallet and approve a $0.01 USDC payment to get your payment proof.
    
    <CodeGroup>
    ```bash Browser (Easiest)
    # Simply visit any endpoint in your browser
    # You'll see a payment UI automatically
    https://api.your-domain.com/v1/markets/trends
    ```

    ```python Python (Web3.py)
    from web3 import Web3
    import requests

    # Connect wallet
    w3 = Web3(Web3.HTTPProvider('https://sepolia.base.org'))
    account = w3.eth.account.from_key('your_private_key')

    # Get payment proof
    # (See full example in Python integration guide)
    ```
    </CodeGroup>

  </Step>

  <Step title="Make Your First Request">
    Use the payment proof to call any endpoint.
    
    <CodeGroup>
    ```bash cURL
    curl -X GET "https://api.your-domain.com/v1/markets/trends?limit=10&timeframe=24h" \
      -H "X-Payment-Proof: YOUR_PAYMENT_PROOF_HERE"
    ```

    ```javascript JavaScript (Fetch)
    const response = await fetch(
      'https://api.your-domain.com/v1/markets/trends?limit=10&timeframe=24h',
      {
        headers: {
          'X-Payment-Proof': 'YOUR_PAYMENT_PROOF_HERE'
        }
      }
    );

    const data = await response.json();
    console.log('Trending categories:', data.trends.trending_categories);
    ```

    ```python Python (Requests)
    import requests

    response = requests.get(
        'https://api.your-domain.com/v1/markets/trends',
        headers={
            'X-Payment-Proof': 'YOUR_PAYMENT_PROOF_HERE'

            'limit': 10,
            'timeframe': '24h'
        }

    data = response.json()
    print('Trending categories:', data['trends']['trending_categories'])
    ```
    </CodeGroup>

## Complete Examples

<AccordionGroup>
  <Accordion title="Python Complete Example" icon="python">
    ```python
    import requests
    from web3 import Web3

    # Configuration
    API_BASE_URL = "https://api.your-domain.com"
    PAYMENT_PROOF = "YOUR_PROOF_HERE"

    def get_market_trends(limit=10, timeframe="24h"):
        """Fetch market trends"""
        response = requests.get(
            f"{API_BASE_URL}/v1/markets/trends",
            headers={"X-Payment-Proof": PAYMENT_PROOF},
            params={"limit": limit, "timeframe": timeframe}
        )

        if response.status_code == 200:
            data = response.json()
            return data['trends']
        elif response.status_code == 402:
            print("Payment required!")
            return None
        else:
            print(f"Error: {response.status_code}")
            return None

    # Use it
    trends = get_market_trends()
    if trends:
        print("Top trending categories:")
        for cat in trends['trending_categories'][:5]:
            print(f"  - {cat['category']}: {cat['count']} mentions")
    ```

  </Accordion>

  <Accordion title="JavaScript Complete Example" icon="js">
    ```javascript
    const API_BASE_URL = "https://api.your-domain.com";
    const PAYMENT_PROOF = "YOUR_PROOF_HERE";

    async function getMarketTrends(limit = 10, timeframe = "24h") {
      try {
        const response = await fetch(
          `${API_BASE_URL}/v1/markets/trends?limit=${limit}&timeframe=${timeframe}`,
          {
            headers: {
              'X-Payment-Proof': PAYMENT_PROOF
            }
          }
        );

        if (response.status === 200) {
          const data = await response.json();
          return data.trends;
        } else if (response.status === 402) {
          console.error('Payment required!');
          return null;
        } else {
          console.error(`Error: ${response.status}`);
          return null;
        }
      } catch (error) {
        console.error('Request failed:', error);
        return null;
      }
    }

    // Use it
    const trends = await getMarketTrends();
    if (trends) {
      console.log('Top trending categories:');
      trends.trending_categories.slice(0, 5).forEach(cat => {
        console.log(`  - ${cat.category}: ${cat.count} mentions`);
      });
    }
    ```

  </Accordion>

    interface MarketTrends {
      success: boolean;
      timeframe: string;
      trends: {
        trending_categories: Array<{
          category: string;
          count: number;
          description: string;
        }>;
        trending_narratives: Array<{
          narrative_id: string;
          item_count: number;

    }

    async function getMarketTrends(
      limit: number = 10,
      timeframe: '1h' | '6h' | '24h' | '7d' = '24h'
    ): Promise<MarketTrends | null> {
      const response = await fetch(
        `https://api.your-domain.com/v1/markets/trends?limit=${limit}&timeframe=${timeframe}`,
        {
          headers: {
            'X-Payment-Proof': process.env.PAYMENT_PROOF!
          }
        }
      );

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }

      return response.json();
    }

## Error Handling

<CodeGroup>
```python Python
try:
    response = requests.get(url, headers=headers)
    
    if response.status_code == 200:
        data = response.json()
        # Process data
    elif response.status_code == 402:
        print("Payment required - get new payment proof")
    elif response.status_code == 429:
        print("Rate limited - wait and retry")
    elif response.status_code == 500:
        print("Server error - retry later")
    else:
        print(f"Unexpected error: {response.status_code}")
        
except requests.exceptions.Timeout:
    print("Request timeout - retry")
except requests.exceptions.ConnectionError:
    print("Connection error - check internet")
```

```javascript JavaScript
try {
  const response = await fetch(url, { headers });

  if (response.status === 200) {
    const data = await response.json();
    // Process data
  } else if (response.status === 402) {
    console.error("Payment required");
  } else if (response.status === 429) {
    console.error("Rate limited");
  } else {
    console.error(`HTTP ${response.status}`);
  }
} catch (error) {
  if (error.name === "AbortError") {
    console.error("Request timeout");
  } else {
    console.error("Request failed:", error);
  }
}
```

</CodeGroup>
